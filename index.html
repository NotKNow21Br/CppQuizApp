Es 1: Riga con somma più alta in una matrice

Spiegazione: Iteriamo per ogni riga e calcoliamo la somma, salvandola se è la massima.


#include <iostream>
using namespace std;

// Funzione che riceve la matrice e trova la riga con somma maggiore
void stampaRigaSommaMassima(int mat[4][4]) {
    int sommaMax = -2147483648; // Inizializziamo con il valore minimo possibile
    int indiceRigaMax = 0;

    for (int i = 0; i < 4; i++) { // Ciclo per ogni riga
        int sommaCorrente = 0;
        for (int j = 0; j < 4; j++) { // Ciclo per ogni colonna
            sommaCorrente += mat[i][j]; // Sommo gli elementi della riga i
        }

        // Se la somma della riga attuale è maggiore della massima trovata finora
        if (sommaCorrente > sommaMax) {
            sommaMax = sommaCorrente;
            indiceRigaMax = i;
        }
    }

    // Stampiamo la riga trovata
    cout << "La riga con la somma piu' alta e' la numero: " << indiceRigaMax << " (somma: " << sommaMax << ")" << endl;
    cout << "Elementi della riga: ";
    for (int j = 0; j < 4; j++) {
        cout << mat[indiceRigaMax][j] << " ";
    }
    cout << endl;
}

Es 2: Parola più lunga in un array di stringhe

Spiegazione: Scorriamo l'array controllando la lunghezza .length() di ogni stringa per trovare il massimo.


#include <iostream>
#include <string>
using namespace std;

// Funzione che ritorna la posizione della stringa più lunga nell'array
int posizioneParolaPiuLunga(string array[], int dimensione) {
    int maxLunghezza = 0;
    int posizione = 0;

    for (int i = 0; i < dimensione; i++) {
        // .length() restituisce il numero di caratteri della stringa
        if (array[i].length() > maxLunghezza) {
            maxLunghezza = array[i].length();
            posizione = i; // Salviamo l'indice (posizione)
        }
    }

    return posizione; 
}

// Esempio d'uso nel main
int main() {
    string parole[] = {"mela", "pomodoro", "casa", "informatica"};
    int pos = posizioneParolaPiuLunga(parole, 4);
    
    cout << "La parola piu' lunga si trova in posizione: " << pos << endl;
    cout << "La parola e': " << parole[pos] << endl;
    
    return 0;
}
